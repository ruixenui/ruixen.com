## Usage

```tsx
import AlertWithActions from "@/components/ruixen/alert-with-actions";

export default function Example() {
  return (
    <AlertWithActions
      variant="error"
      title="Payment failed"
      description="Your card was declined. Update your payment method."
      actions={
        <button className="text-xs font-medium underline underline-offset-2">
          Update billing
        </button>
      }
      onDismiss={() => console.log("dismissed")}
    />
  );
}
```

## Props

| Prop          | Type                                          | Default  | Description                          |
| ------------- | --------------------------------------------- | -------- | ------------------------------------ |
| `variant`     | `"info" \| "success" \| "warning" \| "error"` | `"info"` | Alert variant with matching icon     |
| `title`       | `string`                                      | -        | Alert heading text                   |
| `description` | `string`                                      | -        | Optional description below the title |
| `actions`     | `React.ReactNode`                             | -        | Action buttons or links              |
| `onDismiss`   | `() => void`                                  | -        | Shows dismiss button when provided   |
| `className`   | `string`                                      | -        | Additional CSS classes               |
