## Usage

```tsx
import BannerCookie from "@/components/ruixen/banner-cookie";

export default function MyPage() {
  return (
    <div className="fixed bottom-4 left-4 z-50">
      <BannerCookie
        onAccept={() => console.log("accepted")}
        onDecline={() => console.log("declined")}
      />
    </div>
  );
}
```

## Props

| Prop          | Type         | Default                                                | Description                            |
| ------------- | ------------ | ------------------------------------------------------ | -------------------------------------- |
| `description` | `string`     | `"This site uses cookies to improve your experience."` | The consent message                    |
| `onAccept`    | `() => void` | -                                                      | Callback fired after accept animation  |
| `onDecline`   | `() => void` | -                                                      | Callback fired after decline animation |
| `className`   | `string`     | -                                                      | Additional CSS classes                 |
