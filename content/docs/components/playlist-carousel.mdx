## Usage

```tsx
import PlaylistCarousel, { Track } from "@/components/ruixen/playlist-carousel";

const tracks: Track[] = [
  {
    id: "1",
    title: "Midnight Drive",
    artist: "Neon Pulse",
    cover: "/covers/track1.jpg",
    duration: "3:45",
    src: "/audio/track1.mp3",
  },
  {
    id: "2",
    title: "Ocean Breeze",
    artist: "Calm Shores",
    cover: "/covers/track2.jpg",
    duration: "4:20",
    src: "/audio/track2.mp3",
  },
];

export default function App() {
  return <PlaylistCarousel tracks={tracks} />;
}
```

## Props

| Prop        | Type      | Default | Description            |
| ----------- | --------- | ------- | ---------------------- |
| `tracks`    | `Track[]` | -       | Array of track objects |
| `className` | `string`  | -       | Additional CSS classes |

## Track Type

```typescript
export interface Track {
  id: string;
  title: string;
  artist: string;
  cover: string;
  duration: string;
  src: string;
}
```
