## Usage

```tsx
"use client";

import { useState } from "react";
import AnimatedAlert from "@/components/ruixen/animated-alert";

export default function Example() {
  const [show, setShow] = useState(true);

  return (
    <AnimatedAlert
      variant="success"
      title="Changes saved"
      description="Your preferences have been updated."
      show={show}
      onDismiss={() => setShow(false)}
      autoDismiss={5000}
    />
  );
}
```

## Props

| Prop          | Type                                          | Default  | Description                                   |
| ------------- | --------------------------------------------- | -------- | --------------------------------------------- |
| `variant`     | `"info" \| "success" \| "warning" \| "error"` | `"info"` | Alert variant with matching icon              |
| `title`       | `string`                                      | -        | Alert heading text                            |
| `description` | `string`                                      | -        | Optional description below the title          |
| `actions`     | `React.ReactNode`                             | -        | Action buttons or links                       |
| `onDismiss`   | `() => void`                                  | -        | Callback when alert is dismissed              |
| `autoDismiss` | `number`                                      | -        | Auto-dismiss after ms (shows progress bar)    |
| `show`        | `boolean`                                     | `true`   | Controls visibility with enter/exit animation |
| `className`   | `string`                                      | -        | Additional CSS classes                        |
